average_of_first  = int(input('Enter first score:'))

average_of_second  = int(input('Enter second score:'))

average_of_third  = int(input('Enter third score:'))

average_of_three_score = average_of_first + average_of_second + average_of_third // 2

print('average of three score is',average_of_three_score)

if average_of_three_score <= 90 and average_of_three_score<= 90:
	print('A')

if average_of_three_score <= 80 and average_of_three_score<= 100:
	print('B')

if average_of_three_score <= 70 and average_of_three_score<= 80:
	print('C')

if average_of_three_score <= 60 and average_of_three_score<= 70:
	print('D')

if average_of_three_score <= 0 and average_of_three_score<= 60:
	print('F')
